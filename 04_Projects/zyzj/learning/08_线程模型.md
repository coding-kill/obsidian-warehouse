## 1. C# 线程模型

### async/await 原理

```csharp
// async/await 不是新建线程，而是状态机
// 编译器将 async 方法转换为状态机

// 这个方法不会阻塞线程
public async Task<string> FetchDataAsync()
{
    // 线程 A 执行这里
    var result = await httpClient.GetStringAsync("http://api");  // 线程返回池

    // await 完成后，线程 B (可能是任何线程) 继续执行这里
    return result.ToUpper();
}

// 等价于 Java 的 CompletableFuture
// CompletableFuture<String> future = httpClient.getStringAsync(url)
//     .thenApply(String::toUpperCase);
```

### async vs Task.Run

```csharp
// ❌ 不要用 Task.Run 包装同步代码
public async Task DoWorkAsync()
{
    // 错误：同步代码不需要线程池
    var result = await Task.Run(() => HeavyComputation());  // 多此一举

    // 正确：CPU 密集型用 Task.Run，I/O 密集型用 async
    var result = await Task.Run(() => CalculatePi());  // CPU 密集型 ✓
    var data = await httpClient.GetAsync(url);         // I/O 密集型 ✓
}

// 本项目典型用法：I/O 密集型
public async Task<(eNetResultType, ulong)> UserUid(string id, string authKey)
{
    // await 释放线程，I/O 完成后再续接
    await using var conn = new MySqlConnection(_connString);
    await conn.OpenAsync();
    // ...
}
```

### lock 锁

```csharp
// C# lock = Java synchronized
private readonly object _lock = new object();
private int _counter;

// 安全递增
public void Increment()
{
    lock (_lock)  // 等价于 Java: synchronized(this)
    {
        _counter++;
    }
}

// 本项目典型：保护集合操作
// 来源: ChatServer/Hub/ChatHub.cs
private static readonly object _userListLock = new object();

public void AddUser(UserInfo user)
{
    lock (_userListLock)
    {
        _users.Add(user.UserUid, user);
    }
}

// ❌ lock 的坑：不要 lock 字符串、this
string lockKey = "user_123";  // 错误！字符串驻留，可能死锁
lock (this) { }  // 错误！外部可能也 lock(this)
```

### Interlocked 原子操作

```csharp
// 原子操作，比 lock 更轻量
private int _counter;

// 等价于 lock (_lock) { _counter++; }
Interlocked.Increment(ref _counter);

// 等价于 lock (_lock) { _counter = 0; }
Interlocked.Exchange(ref _counter, 0);

// 本项目典型用法
// 来源: LoginController.cs
private static int _requestCount = 0;

[HttpPost("TestLogin")]
public async Task<object> TestLogin(...)
{
    var a = Interlocked.Increment(ref _requestCount);  // 原子递增
    log.Debug($"TestLogin. Count: {a}");
    // ...
}
```

### ConcurrentDictionary 线程安全集合

```csharp
// 线程安全的字典，无需 lock
// 来源: ChatServer/Hub/ChatHub.cs

// key-value 操作原子化
ConcurrentDictionary<ulong, UserInfo> _users = new();

// 添加（线程安全）
_users.TryAdd(userUid, userInfo);

// 移除（线程安全）
_users.TryRemove(userUid, out var removed);

// 查询（线程安全）
_users.TryGetValue(userUid, out var user);

// 本项目典型模式：先检查后操作
if (_users.TryGetValue(userUid, out var user))
{
    // 此时用户可能已被其他线程移除！
    // → 需要组合操作，用 lock 或改用其他数据结构
}
```

### 常见并发问题

```csharp
// ❌ 竞态条件 (Race Condition)
private int _balance = 100;

// 线程 A                    // 线程 B
// read _balance (100)       // read _balance (100)
// subtract 50 (50)           // subtract 30 (70)
// write _balance (50)       // write _balance (70)  ← 最终余额错误！

// ✅ 解决方案1: lock
private readonly object _lock = new object();
private int _balance = 100;

public void Withdraw(int amount)
{
    lock (_lock)
    {
        if (_balance >= amount)
        {
            _balance -= amount;  // 原子读-改-写
        }
    }
}

// ✅ 解决方案2: Interlocked (仅限简单操作)
private int _balance = 100;

public void Withdraw(int amount)
{
    // 循环直到成功（基于 CAS）
    while (true)
    {
        int current = _balance;
        if (current < amount)
            throw new Exception("余额不足");
        if (Interlocked.CompareExchange(ref _balance, current - amount, current) == current)
            return;  // 成功
    }
}

// ✅ 解决方案3: SemaphoreSlim (限流)
private static SemaphoreSlim _semaphore = new SemaphoreSlim(10);  // 最多10并发

public async Task DoWorkAsync()
{
    await _semaphore.WaitAsync();  // 获取信号量
    try
    {
        // 并发控制
    }
    finally
    {
        _semaphore.Release();  // 释放
    }
}
```

## 2. C++ 线程模型

### CRITICAL_SECTION 互斥锁

```cpp
// C++ Windows 临界区 = C# lock
// 来源: CommonLib/T3Synchronized.h

class Synchronized {
public:
    // 定义锁对象
    class SectionObject {
    public:
        SectionObject() {
            ::InitializeCriticalSection(&cs);  // 初始化
        }
        ~SectionObject() {
            ::DeleteCriticalSection(&cs);     // 销毁
        }
        void enter() { ::EnterCriticalSection(&cs); }
        void leave() { ::LeaveCriticalSection(&cs); }
    private:
        CRITICAL_SECTION cs;
    };

    // 构造时加锁，析构时解锁（RAII）
    Synchronized(SectionObject* so) : m_sectionObject(so) {
        if (m_sectionObject)
            m_sectionObject->enter();
    };
    ~Synchronized() {
        if (m_sectionObject)
            m_sectionObject->leave();
    };
};
```

### 宏封装（更简洁的用法）

```cpp
// 定义锁
CRITICAL_SECTION m_lock;

// 使用方式1：手动
void DoWork() {
    EnterCriticalSection(&m_lock);
    try {
        // 临界区代码
    } finally {
        LeaveCriticalSection(&m_lock);
    }
}

// 使用方式2：RAII 封装（推荐）
#define AUTO_LOCK(lock) std::unique_lock<decltype(lock)> l##lock {lock};

// 来源: CommonLib/T3Synchronized.h
#define AUTO_LOCK(lock)  std::unique_lock<decltype(lock)> l##lock {lock};

// 使用
void DoWork() {
    AUTO_LOCK(m_lock);  // 自动加锁，作用域结束时自动释放
    // 临界区代码
}
```

### std::mutex (C++11 跨平台)

```cpp
#include <mutex>

class ThreadSafeCounter {
private:
    int _counter = 0;
    std::mutex _mutex;

public:
    void Increment() {
        std::lock_guard<std::mutex> lock(_mutex);  // RAII 自动锁
        _counter++;
    }

    int Get() {
        std::lock_guard<std::mutex> lock(_mutex);
        return _counter;
    }
};
```

### atomic 原子操作

```cpp
// C++ 原子变量 = C# Interlocked
// 来源: CommonLib/T3Session.h

std::atomic<DWORD> m_pushCount = 0;      // 原子计数器
std::atomic<bool> m_bSending = false;   // 原子布尔

// 使用
m_pushCount.fetch_add(1);        // 等价 Interlocked.Increment
m_bSending.store(true);          // 写
bool flag = m_bSending.load();   // 读
```

### 读写锁 (shared_mutex)

```cpp
// C++17 读写锁
// 来源: CommonLib/T3Synchronized.h

using RWLock = std::shared_mutex;

// 读锁：共享访问
std::shared_lock<std::shared_mutex> readLock(rwLock_);
// 多个线程可以同时读

// 写锁：独占访问
std::unique_lock<std::shared_mutex> writeLock(rwLock_);
// 独占，写操作时不能有其他读写

// 本项目使用宏封装
#define READ_LOCK(lock)  std::shared_lock<decltype(lock)> l##lock {lock};
#define AUTO_LOCK(lock) std::unique_lock<decltype(lock)> l##lock {lock};

// 使用
void Read() {
    READ_LOCK(m_rwLock);    // 多个线程可同时读
    // 读操作
}

void Write() {
    AUTO_LOCK(m_rwLock);    // 独占写
    // 写操作
}
```

### 递归锁

```cpp
// 递归锁：同一线程可多次获取
// 来源: CommonLib/T3Synchronized.h

using RCLock = std::recursive_mutex;

RCLock m_recursiveLock;

void FuncA() {
    AUTO_LOCK(m_recursiveLock);
    FuncB();  // 可重入调用 FuncB
}

void FuncB() {
    AUTO_LOCK(m_recursiveLock);  // 同一线程可再次获取
}
```

## 3. 发送队列 (SendFunc)

### 为什么需要发送队列

```cpp
// 问题：多线程同时发送会冲突
// 线程1: send(packet1)  线程2: send(packet2)
// 可能交替执行，导致数据包混淆

// 解决方案：每个 Session 维护发送队列
// 串行化发送操作，保证数据完整性
```

### Session 发送队列

```cpp
// 来源: CommonLib/T3Session.h

class T3Session {
private:
    // 发送队列
    CircularQueue<Packet*>  m_sendPacketQueue;   // 包队列
    CircularQueue<char>     m_sendBuffer;         // 字节缓冲（二选一）

    // 原子标记
    std::atomic<bool>       m_bSending = false;   // 是否正在发送

    // 发送锁
    mutable RCLock          m_isSendingSec;        // 递归锁
};
```

### 发送流程

```cpp
// 发送流程

// 1. 添加到队列（非阻塞）
bool T3Session::add(Packet* pPacket)
{
    AUTO_LOCK(m_isSendingSec);  // 加锁保护

    // 加入队列
    m_sendPacketQueue.push(pPacket);

    // 如果没有正在发送，启动发送
    if (m_bSending.compare_exchange(false, true) == false)
    {
        // 已经是 true，说明有其他线程在发送
        return true;
    }

    // 自己启动发送
    send();
    return true;
}

// 2. 实际发送（可能跨线程）
DWORD T3Session::send()
{
    while (true)
    {
        AUTO_LOCK(m_isSendingSec);

        // 从队列取包
        Packet* pPacket = m_sendPacketQueue.pop();
        if (!pPacket) break;  // 队列空

        // WSASend 异步发送
        postSend(buffer, size);

        // 继续循环，直到队列空
    }

    m_bSending = false;  // 发送完成
    return 0;
}
```

### GridMulticast 实现

```cpp
// 来源: GameServer/SendFunc.cpp

void SendFunc::GridMulticast(iActor* actor, Packet* packet, bool isSendToMe, int max)
{
    std::unordered_set<User*> users;
    GetAroundUserList(users, actor->GetTerrain(), actor);

    int32_t count = max;
    sendUser(users, packet, count);  // 批量发送
}

bool SendFunc::sendUser(std::unordered_set<User*>& userList, Packet* packet, int32_t& count)
{
    for (auto pUser : userList)
    {
        if (!pUser) continue;

        pUser->send(packet);  // 每个 User 内部处理发送队列

        if (--count <= 0)
            break;  // 达到上限
    }
    return true;
}
```

### 线程安全保证

```cpp
// 发送的线程安全由 Session 内部保证：

// 1. 每个 Session 有独立的锁
// 2. 使用 atomic 标记避免重复发送
// 3. 发送队列是 CircularQueue（线程安全实现）

// Session.add() 可被多个线程并发调用
// 最终通过锁保证串行发送
```

### IOCP 工作线程处理

```cpp
// 工作线程函数（简化）

DWORD WINAPI WorkerThread(LPVOID lpParam)
{
    HANDLE hCompletionPort = (HANDLE)lpParam;

    while (true)
    {
        DWORD bytesTransferred;
        ULONG_PTR key;           // 关联的 Session
        OVERLAPPED* overlap;

        // 等待完成事件（阻塞）
        BOOL result = GetQueuedCompletionStatus(
            hCompletionPort,
            &bytesTransferred,
            &key,
            &overlap,
            INFINITE
        );

        if (overlap == NULL)  // 收到退出信号
            break;

        // 获取 Session
        Session* session = (Session*)key;

        // 根据 overlap 类型分发
        if (overlap->mode == OVERLAPPED_READ)
        {
            // 处理收到的数据
            session->recvProcess(packet);
        }
        else if (overlap->mode == OVERLAPPED_WRITE)
        {
            // 发送完成
            session->onSendComplete(bytesTransferred);
        }
    }

    return 0;
}
```

```cpp
template<typename T>
class SingleTon {
public:
    static T* GetInstance() {
        if (m_pInstance == nullptr) {
            AUTO_LOCK(m_lock);  // 双重检查锁定
            if (m_pInstance == nullptr) {
                m_pInstance = new T();
            }
        }
        return m_pInstance;

    static RCLock m_lock;
```

### C++ 线程安全的消息处理

```cpp
// 来源: GameServer/Actor.cpp

void Actor::AddBuff(Buff* pBuff)
{
    AUTO_LOCK(m_buffLock);  // 保护 buff 列表

    // 检查是否已存在
```

### C# 和 C++ 线程对比

| 场景 | C# | C++ |
|------|-----|-----|
| 互斥锁 | `lock(_lock)` | `CRITICAL_SECTION` / `std::mutex` |
| 原子操作 | `Interlocked.Increment()` | `std::atomic<T>` |
| 线程安全集合 | `ConcurrentDictionary` | `std::unordered_map` + 锁 |
| 读写锁 | `ReaderWriterLockSlim` | `std::shared_mutex` |
| 线程池 | `ThreadPool.QueueUserWorkItem` | `CreateThread` / IOCP |
| 异步 I/O | `async/await` | Overlapped + IOCP |
